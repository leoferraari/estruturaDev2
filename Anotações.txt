CREATE SCHEMA TRABALHOFINAL;

CREATE TABLE TRABALHOFINAL.TBCATEGORIA (
 CATCODIGO SERIAL NOT NULL,
 CATNOME VARCHAR (15) NOT NULL CONSTRAINT UK_CATNOME
UNIQUE,
 CATDESCRICAO VARCHAR,
 CONSTRAINT PK_TBCATEGORIA
 PRIMARY KEY (CATCODIGO) 
);

CREATE TABLE TRABALHOFINAL.TBFORNECEDORES (
 FORCODIGO SERIAL NOT NULL,
 FORNOMECOMPANHIA VARCHAR(40) NOT NULL,
 FORNOMECONTATO VARCHAR(30) NOT NULL,
 FORTITULOCONTATO VARCHAR(30) NOT NULL,
 FORENDERECO VARCHAR(60) NOT NULL,
 FORCIDADE VARCHAR(15) NOT NULL,
 FORREGIAO VARCHAR(15) NOT NULL,
 FORCEP VARCHAR(10) NOT NULL,
 FORPAIS VARCHAR(15) NOT NULL,
 FORTELEFONE VARCHAR(24) NOT NULL,
 FORFAX VARCHAR(24) NOT NULL,
 FORWEBSITE VARCHAR(60) NOT NULL,
 CONSTRAINT PK_TBFORNECEDORES
 PRIMARY KEY (FORCODIGO)
);


CREATE TABLE TRABALHOFINAL.TBPRODUTOS (
 PROCODIGO SERIAL NOT NULL,
 PRONOME VARCHAR (40) NOT NULL,
 PROQTDUNIDADE NUMERIC(17,2) NOT NULL,
 PROPRECOUNITARIO NUMERIC(17,2) NOT NULL,
 PROUNDESTOQUE SMALLINT NOT NULL,
 PROUNDEMORDEM SMALLINT NOT NULL,
 PRONIVELREPOSICAO SMALLINT NOT NULL,
 PRODESCONTINUADO SMALLINT,
 CATCODIGO INTEGER NOT NULL,
 FORCODIGO INTEGER NOT NULL,
 CONSTRAINT PK_TBPRODUTOS
 PRIMARY KEY (PROCODIGO),
 CONSTRAINT "FK TBPRODUTOS -> TBCATEGORIA"
 FOREIGN KEY(CATCODIGO)
 REFERENCES TRABALHOFINAL.TBCATEGORIA(CATCODIGO),
 CONSTRAINT "FK TBPRODUTOS-> TBFORNECEDORES"
 FOREIGN KEY (FORCODIGO)
 REFERENCES TRABALHOFINAL.TBFORNECEDORES(FORCODIGO)
);



CREATE TABLE TRABALHOFINAL.TBCLIENTES (
 CLICODIGO SERIAL NOT NULL,
 CLINOMECOMPANHIA VARCHAR(40) NOT NULL,
 CLINOMECONTATO VARCHAR(30) NOT NULL,
 CLITITULOCONTATO VARCHAR(30) NOT NULL,
 CLIENDERECO VARCHAR(60) NOT NULL,
 CLICIDADE VARCHAR(15) NOT NULL,
 CLIREGIAO VARCHAR(15) NOT NULL,
 CLICEP VARCHAR(10) NOT NULL,
 CLIPAIS VARCHAR(15) NOT NULL,
 CLITELEFONE VARCHAR(24) NOT NULL,
 CLIFAX VARCHAR(24) NOT NULL,
 CONSTRAINT PK_TBCLIENTES
 PRIMARY KEY (CLICODIGO) 
);

CREATE TABLE TRABALHOFINAL.TBUSUARIOS
(
  usucodigo serial NOT NULL,
  usuome character varying(60) NOT NULL,
  usuemail character varying(35) NOT NULL,
  ususenha character varying(32) NOT NULL,
  PRIMARY KEY (usucodigo),
  CONSTRAINT uk_usuemail UNIQUE (usuemail)
)
